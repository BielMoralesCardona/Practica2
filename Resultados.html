datos<-data %>% select(28:37) %>% 
  mutate(across(everything(), ~case_when(
     . == "Mai" ~ 0,
     . == "Gairebé mai" ~ 1,
     . == "A vegades" ~ 2,
     . == "Gairebé sempre" ~ 3,
     . == "Sempre" ~ 4
   ))
   )

datos<-datos %>% mutate(VHI=rowSums(datos))
